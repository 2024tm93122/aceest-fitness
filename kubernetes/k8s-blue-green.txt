# Blue-Green Deployment Strategy
# Blue Environment (Current Production)
apiVersion: apps/v1
kind: Deployment
metadata:
  name: aceest-fitness-blue
  labels:
    app: aceest-fitness
    environment: blue
spec:
  replicas: 3
  selector:
    matchLabels:
      app: aceest-fitness
      environment: blue
  template:
    metadata:
      labels:
        app: aceest-fitness
        environment: blue
    spec:
      containers:
      - name: aceest-fitness
        image: docker.io/aceest-fitness:v1.2.3
        ports:
        - containerPort: 8000
        livenessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 8000
          initialDelaySeconds: 10
          periodSeconds: 5
        resources:
          requests:
            memory: "128Mi"
            cpu: "100m"
          limits:
            memory: "256Mi"
            cpu: "200m"
---

# Service - Switch between blue and green by changing selector
apiVersion: v1
kind: Service
metadata:
  name: aceest-fitness-service
  labels:
    app: aceest-fitness
spec:
  type: LoadBalancer
  selector:
    app: aceest-fitness
    environment: blue  # Change to 'green' to switch traffic
  ports:
  - port: 8000
    targetPort: 8000
    protocol: TCP
    name: http

# To switch to green deployment, run:
# kubectl patch service aceest-fitness-service -p '{"spec":{"selector":{"environment":"green"}}}'

# To rollback to blue deployment, run:
# kubectl patch service aceest-fitness-service -p '{"spec":{"selector":{"environment":"blue"}}}'
